#include < stdio.h>
#include < math.h>

double function(double x) {
    return x * x - 4;
}

void bisection(double a, double b, double tol) {
    if (function(a) * function(b) >= 0) {
        printf("Bisection method fails. f(a) and f(b) must have different signs.\n");
        return;
    }

    double c;
    while ((b - a) >= tol) {
        c = (a + b) / 2;

        if (function(c) == 0.0) {
            break;
        }
        if (function(c) * function(a) < 0) {
            b = c;
        } else {
            a = c;
        }
    }
    printf("The root is: %.5f\n", c);
}

int main() {
    double a, b, tol;

    printf("Enter the interval [a, b]: ");
    scanf("%lf %lf", &a, &b);
    printf("Enter the tolerance: ");
    scanf("%lf", &tol);

    bisection(a, b, tol);

    return 0;
}

#include < stdio.h>
#include < math.h>

double function(double x) {
    return x * x - 4;
}

void regulaFalsi(double a, double b, double tol) {
    if (function(a) * function(b) >= 0) {
        return;
    }

    double c;
    while (1) {
        c = (a * function(b) - b * function(a)) / (function(b) - function(a));
        if (fabs(function(c)) <= tol) break;

        if (function(c) * function(a) < 0) {
            b = c;
        } else {
            a = c;
        }
    }

    printf("The root is: %.5f\n", c);
}

int main() {
    double a, b, tol;
    printf("Enter the interval [a, b]: ");
    scanf("%lf %lf", &a, &b);
    printf("Enter the tolerance: ");
    scanf("%lf", &tol);
    regulaFalsi(a, b, tol);
    return 0;
}

#include < stdio.h>
#include < math.h>

    return x * x - 4;
}

void secantMethod(double x0, double x1, double tol) {
    double x2;

    while (fabs(function(x1)) > tol) {
        x2 = x1 - (function(x1) * (x1 - x0)) / (function(x1) - function(x0));
        x0 = x1;
        x1 = x2;
    }

    printf("The root is: %.5f\n", x2);
}

    #include < stdio.h>

        #define MAX 10
        
        void gaussElimination(float a[MAX][MAX], float b[MAX], int n) {
            for (int i = 0; i < n; i++) {
                // Partial pivoting
                for (int k = i + 1; k < n; k++) {
                    if (a[k][i] > a[i][i]) {
                        for (int j = 0; j < n; j++) {
                            float temp = a[i][j];
                            a[i][j] = a[k][j];
                            a[k][j] = temp;
                        }
                        float temp = b[i];
                        b[i] = b[k];
                        b[k] = temp;
                    }
                }
        
                // Forward elimination
                for (int k = i + 1; k < n; k++) {
                    float factor = a[k][i] / a[i][i];
                    for (int j = i; j < n; j++) {
                        a[k][j] -= factor * a[i][j];
                    }
                    b[k] -= factor * b[i];
                }
            }
        
            // Back substitution
            float x[MAX];
            for (int i = n - 1; i >= 0; i--) {
                x[i] = b[i];
                for (int j = i + 1; j < n; j++) {
                    x[i] -= a[i][j] * x[j];
                }
                x[i] /= a[i][i];
            }
        
            // Display the results
            printf("Solutions:\n");
            for (int i = 0; i < n; i++) {
                printf("x%d = %.2f\n", i + 1, x[i]);
            }
        }
        
        int main() {
            int n;
            float a[MAX][MAX], b[MAX];
        
            printf("Enter the number of equations: ");
            scanf("%d", &n);
            printf("Enter the coefficients of the equations:\n");
            for (int i = 0; i < n; i++) {
                for (int j = 0; j < n; j++) {
                    scanf("%f", &a[i][j]);
                }
                scanf("%f", &b[i]); // Right-hand side
            }
        
            gaussElimination(a, b, n);
        
            return 0;
        }

        #define MAX 10
        
        void birgeVieta(float coeff[MAX], int degree, float initialGuess, float tolerance) {
            float x = initialGuess;
            float f, f_prime;
            int iteration = 0;
        
            do {
                // Calculate the polynomial value and its derivative using synthetic division
                float p[MAX] = {0}; // Array to hold synthetic division results
                p[0] = coeff[0];
        
                for (int i = 1; i <= degree; i++) {
                    p[i] = coeff[i] + x * p[i - 1];
                }
        
                f = p[degree];
                f_prime = p[degree - 1];
        
                // Update the guess using Birge-Vieta formula
                x = x - f / f_prime;
        
                printf("Iteration %d: x = %.6f, f(x) = %.6f\n", iteration++, x, f);
        
            } while (fabs(f) > tolerance && iteration < 100); // Max 100 iterations
        
            printf("Approximate root: %.6f\n", x);
        }

    #include < stdio.h>
        #include < math.h>
        
        #define MAX 10
        
        void gaussJordan(float a[MAX][MAX], float b[MAX], int n) {
            // Forward elimination
            for (int i = 0; i < n; i++) {
                // Pivoting
                for (int k = i + 1; k < n; k++) {
                    if (fabs(a[k][i]) > fabs(a[i][i])) {
                        for (int j = 0; j < n; j++) {
                            float temp = a[i][j];
                            a[i][j] = a[k][j];
                            a[k][j] = temp;
                        }
                        float temp = b[i];
                        b[i] = b[k];
                        b[k] = temp;
                    }
                }
        
                // Normalize the pivot row
                float pivot = a[i][i];
                for (int j = 0; j < n; j++) {
                    a[i][j] /= pivot;
                }
                b[i] /= pivot;
        
                // Eliminate all other entries in this column
                for (int k = 0; k < n; k++) {
                    if (k != i) {
                        float factor = a[k][i];
                        for (int j = 0; j < n; j++) {
                            a[k][j] -= factor * a[i][j];
                        }
                        b[k] -= factor * b[i];
                    }
                }
            }
        
            printf("Solutions:\n");
            for (int i = 0; i < n; i++) {
                printf("x%d = %.6f\n", i + 1, b[i]);
            }
        }
        
        int main() {
            int n;
            float a[MAX][MAX], b[MAX];
        
            printf("Enter the number of equations: ");

            printf("Enter the coefficients of the equations (augmented matrix):\n");

    #include < stdio.h>
        #include < math.h>
        
        #define MAX 10
        
        void gaussSeidel(float a[MAX][MAX], float b[MAX], float x[MAX], int n, float tolerance) {
            float error;
            int iterations = 0;
        
            do {
                error = 0.0;
                for (int i = 0; i < n; i++) {
                    float sum = 0.0;
        
                    for (int j = 0; j < n; j++) {
                        if (j != i) {
                            sum += a[i][j] * x[j];
                        }
                    }
        
                    float newX = (b[i] - sum) / a[i][i];
                    error = fmax(error, fabs(newX - x[i])); // Track the maximum error
                    x[i] = newX; // Update the solution
                }
                iterations++;
            } while (error > tolerance);
        
            printf("Converged after %d iterations:\n", iterations);
            for (int i = 0; i < n; i++) {
                printf("x%d = %.6f\n", i + 1, x[i]);
            }
        }

        double lagrangeInterpolation(double x[], double y[], int n, double xValue) {
            double result = 0.0;
        
            for (int i = 0; i < n; i++) {
                double term = y[i];
                for (int j = 0; j < n; j++) {
                    if (j != i) {
                        term *= (xValue - x[j]) / (x[i] - x[j]);
                    }
                }
                result += term;
            }
        
            return result;
        }

        #define MAX 10
        
        void computeForwardDifference(float x[], float y[][MAX], int n) {
            for (int j = 1; j < n; j++) {
                for (int i = 0; i < n - j; i++) {
                    y[i][j] = y[i + 1][j - 1] - y[i][j - 1];
                }
            }
        }
        
        float newtonForwardInterpolation(float x[], float y[][MAX], int n, float value) {
            float h = x[1] - x[0];
            float result = y[0][0];
            float u = (value - x[0]) / h;
            
            for (int i = 1; i < n; i++) {
                float term = 1;
                for (int j = 0; j < i; j++) {
                    term *= (u - j) / (i);
                }
                result += term * y[0][i];
            }
        
            return result;
        }

        void computeBackwardDifference(float x[], float y[][MAX], int n) {
            for (int j = 1; j < n; j++) {
                for (int i = n - 1; i >= j; i--) {
                    y[i][j] = y[i][j - 1] - y[i - 1][j - 1];
                }
            }
        }
        
        float newtonBackwardInterpolation(float x[], float y[][MAX], int n, float value) {
            float h = x[1] - x[0];
            float result = y[n - 1][0];
            float u = (value - x[n - 1]) / h;
        
            for (int i = 1; i < n; i++) {
                float term = 1;
                for (int j = 0; j < i; j++) {
                    term *= (u + j) / (i);
                }
                result += term * y[n - 1 - i][i];
            }
        
            return result;
        }

    #include < stdio.h>

        // Function to be integrated
        float f(float x) {
            return x * x; // Example function: f(x) = x^2
        }
        
        // Trapezoidal rule implementation
        float trapezoidal(float a, float b, int n) {
            float h = (b - a) / n; // Width of each trapezoid
            float sum = 0.5 * (f(a) + f(b)); // Sum of the first and last terms
        
            for (int i = 1; i < n; i++) {
                sum += f(a + i * h); // Add the middle terms
            }
        
            return sum * h; // Final result
        }
